47  for (int i=0; i<y.size(); i++) yt[i] = y[i] + h*k3[i];// Preparing fourth step yt <- y + k3
48  derivs(x+h, yt, k4); // Final step : evaluating k4
49  for (int i=0; i<y.size(); i++) y[i] += h/6.0*(k1[i]+2.0*(k2[i]+k3[i])+k4[i]);
50 }

145 // Fifth-order Runge-Kutta step with monitoring of local truncation error to ensure accuracy and
146 // adjust stepsize. Input are
147 // y[...], dydx[...] -- the dependent variable vector and its derivative at the starting value of the independent variable x.
148 // htry, hdid -- the stepsize to be attempted htry, and the stepsize that was actually accomplished hdid
149 // hnext -- the estimated next stepsize,
150 // derivs -- the user-supplied routine that computes the right-hand side derivatives.
151 //
152 // Also used in the class are the required accuracy eps, and the vector yscal[...] against which the error is scaled.
153 // On output, y and x are replaced by their new values
154 template <class container>
155 template <class functor>
156 void RKCK45<container>::RKStep(container& y, container& dydx, double& x, double htry,
157  double& hdid, double& hnext, functor& derivs){
158 
159  container yerr(y.size()), ytemp(y.size()); // TODO: Should these also be made class members to prevent reallocation?
160  double errmax;
161  double h=htry; // Set stepsize to the initial trial value.
162  for (;;) { // infinite loop
163  RKTry(y, dydx, x, h, ytemp, yerr, derivs);// Take a step.
164  errmax=0.0; // Evaluate accuracy.
165  for (int i=0; i<y.size(); i++) errmax=std::max(errmax,fabs(yerr[i]/yscal[i]));
166  errmax /= eps; // Scale relative to required tolerance.
167  if (errmax <= 1.0) break; // Step succeeded. Compute size of next step.
168  double htemp=SAFETY*h*pow(errmax,PSHRNK); // Truncation error too large, reduce stepsize.
169  h=(h >= 0.0 ? std::max(htemp,0.1*h) : std::min(htemp,0.1*h)); // No more than a factor of 10.
170  if ((x+h) == x) std::cerr<<"stepsize underflow in rkqs"<<std::endl;
171  }
172  if (errmax > ERRCON) hnext=SAFETY*h*pow(errmax,PGROW); // Step is too small, increase it next time
173  else hnext=5.0*h; // No more than a factor of 5 increase.
174  hdid=h;
175  x += h;
176  for (int i=0; i<y.size();i++) y[i] = ytemp[i];
177 }
