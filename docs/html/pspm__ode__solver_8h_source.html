99 void VRungeKutta<container>::RKStep(container& y, container& dydx, double& x, double htry,
100  double& hdid, double& hnext, functor& derivs)
101 //Fifth-order Runge-Kutta step with monitoring of local truncation error to ensure accuracy and
102 // adjust stepsize. Input are
103 // y[...], dydx[...] -- the dependent variable vector and its derivative at the starting value of the independent variable x.
104 // htry, hdid -- the stepsize to be attempted htry, and the stepsize that was actually accomplished hdid
105 // hnext -- the estimated next stepsize,
106 // derivs -- the user-supplied routine that computes the right-hand side derivatives.
107 //
108 // Also used in the class are the required accuracy eps, and the vector yscal[...] against which the error is scaled.
109 // On output, y and x are replaced by their new values
110 {
111  container yerr(y.size()), ytemp(y.size());
112  double errmax;
113  double h=htry; // Set stepsize to the initial trial value.
114  for (;;) { // infinite loop
115  RKTry(y, dydx, x, h, ytemp, yerr, derivs);// Take a step.
116  errmax=0.0; // Evaluate accuracy.
117  for (int i=0; i<y.size(); i++) errmax=std::max(errmax,fabs(yerr[i]/yscal[i]));
118  errmax /= eps; // Scale relative to required tolerance.
119  if (errmax <= 1.0) break; // Step succeeded. Compute size of next step.
120  double htemp=SAFETY*h*pow(errmax,PSHRNK); // Truncation error too large, reduce stepsize.
121  h=(h >= 0.0 ? std::max(htemp,0.1*h) : std::min(htemp,0.1*h)); // No more than a factor of 10.
122  if ((x+h) == x) std::cerr<<"stepsize underflow in rkqs"<<std::endl;
123  }
124  if (errmax > ERRCON) hnext=SAFETY*h*pow(errmax,PGROW); // Step is too small, increase it next time
125  else hnext=5.0*h; // No more than a factor of 5 increase.
126  hdid=h;
127  x += h;
128  for (int i=0; i<y.size();i++) y[i] = ytemp[i];
129 }
